Private Function CodePuzzle(sList As String) As String

    Dim r As Integer
    Dim x As Integer
    Dim c As Integer


    x = Int(Rnd(-7))

    For r = 1 To Len(sList)

        x = Int(Rnd() * 96)
        c = Asc(Mid(sList, r, 1))
        c = c + x
        If c >= 126 Then c = c - 126 + 32

    CodePuzzle = CodePuzzle & Chr$(c)

    Next

End Function
